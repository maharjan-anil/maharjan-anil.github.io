    async function incrementVisitorCount() {
        if (!db || sessionStorage.getItem('visitorCounted')) return;
        
        const counterRef = db.collection('stats').doc('visitors');
        
        try {
            // Use atomic increment to avoid race conditions
            await counterRef.set({
                count: firebase.firestore.FieldValue.increment(1)
            }, { merge: true });
            
            // Mark this session as counted
            sessionStorage.setItem('visitorCounted', 'true');
            
            // Get and display the updated count
            const doc = await counterRef.get();
            if (doc.exists) {
                document.getElementById('visitorCount').textContent = doc.data().count.toLocaleString();
            }
        } catch (error) {
            console.error('Error updating visitor count:', error);
            // Fallback: try to create the document if it doesn't exist
            if (error.code === 'not-found') {
                await counterRef.set({ count: 1 });
                document.getElementById('visitorCount').textContent = '1';
                sessionStorage.setItem('visitorCounted', 'true');
            }
        }
    }
